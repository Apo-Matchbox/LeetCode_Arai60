```cpp
#include <iostream>
#include <set>
using namespace std;

struct ListNode {
    int val;
    ListNode* next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
    public:
        bool hasCycle(ListNode* head) {
            set<ListNode*> reached;
            ListNode* current = head;
            while (current) {
                if(reached.contains(current)) {
                    return true;
                }
                reached.insert(current);
                current = current->next;
            }
            return false;
        }
};
```
